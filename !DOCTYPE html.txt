<section id="nota">
  <h2>📚 Nota Kata Adjektif</h2>

  <h3>Apa itu Kata Adjektif?</h3>
  <p>
    <b>Kata adjektif</b> ialah perkataan yang menerangkan 
    <span class="highlight">sifat, keadaan atau ciri</span> sesuatu kata nama.
  </p>

  <div class="info-box">
    <p><b>Formula mudah:</b></p>
    <p>
      Kata Nama + Kata Adjektif
    </p>
    <p>
      Contoh: rumah <b>besar</b>, bunga <b>cantik</b>, air <b>panas</b>
    </p>
  </div>

  <h3>Jenis-jenis Kata Adjektif</h3>

  <div class="nota-card">
    <div class="card c1">
      <h3>1. Sifat / Keadaan</h3>
      <p>Menerangkan sifat seseorang, haiwan atau benda.</p>
      <p><b>Contoh:</b> baik, rajin, pandai, cantik, kuat</p>
      <p><b>Ayat:</b> Amir murid yang <b>rajin</b>.</p>
    </div>

    <div class="card c2">
      <h3>2. Warna</h3>
      <p>Menerangkan warna sesuatu benda.</p>
      <p><b>Contoh:</b> merah, biru, hijau, kuning, putih</p>
      <p><b>Ayat:</b> Baju itu berwarna <b>biru</b>.</p>
    </div>

    <div class="card c3">
      <h3>3. Ukuran</h3>
      <p>Menerangkan saiz atau ukuran.</p>
      <p><b>Contoh:</b> besar, kecil, panjang, pendek, tinggi</p>
      <p><b>Ayat:</b> Pokok itu sangat <b>tinggi</b>.</p>
    </div>

    <div class="card c4">
      <h3>4. Bentuk</h3>
      <p>Menerangkan rupa bentuk sesuatu.</p>
      <p><b>Contoh:</b> bulat, bujur, leper, lonjong</p>
      <p><b>Ayat:</b> Bola itu berbentuk <b>bulat</b>.</p>
    </div>

    <div class="card c5">
      <h3>5. Waktu</h3>
      <p>Menerangkan masa atau tempoh.</p>
      <p><b>Contoh:</b> awal, lewat, lama, baharu</p>
      <p><b>Ayat:</b> Siti datang <b>awal</b> ke sekolah.</p>
    </div>

    <div class="card c6">
      <h3>6. Jarak</h3>
      <p>Menerangkan jauh atau dekat sesuatu tempat.</p>
      <p><b>Contoh:</b> jauh, dekat, hampir</p>
      <p><b>Ayat:</b> Rumah saya <b>dekat</b> dengan sekolah.</p>
    </div>

    <div class="card c7">
      <h3>7. Perasaan</h3>
      <p>Menerangkan emosi seseorang.</p>
      <p><b>Contoh:</b> gembira, sedih, marah, takut</p>
      <p><b>Ayat:</b> Adik berasa <b>gembira</b>.</p>
    </div>

    <div class="card c8">
      <h3>8. Cara</h3>
      <p>Menerangkan gaya atau cara sesuatu dilakukan.</p>
      <p><b>Contoh:</b> laju, perlahan, cepat, lambat</p>
      <p><b>Ayat:</b> Kereta itu bergerak <b>laju</b>.</p>
    </div>

    <div class="card c9">
      <h3>9. Pancaindera</h3>
      <p>Menerangkan rasa, bau, bunyi atau sentuhan.</p>
      <p><b>Contoh:</b> manis, masin, wangi, busuk, kasar</p>
      <p><b>Ayat:</b> Kuih itu sangat <b>manis</b>.</p>
    </div>
  </div>

  <h3>Nota Tambahan</h3>

  <details>
    <summary>Kata penguat dengan kata adjektif</summary>
    <p>
      Kata adjektif boleh disertai kata penguat seperti 
      <b>sangat, amat, paling, terlalu</b>.
    </p>
    <p>Contoh: Rumah itu <b>sangat besar</b>.</p>
  </details>

  <details>
    <summary>Contoh kata adjektif dalam ayat</summary>
    <ul>
      <li>Buah mangga itu <b>manis</b>.</li>
      <li>Cuaca hari ini sangat <b>panas</b>.</li>
      <li>Meja itu <b>kecil</b>.</li>
      <li>Sarah berasa <b>sedih</b>.</li>
      <li>Jalan ke kampung itu agak <b>jauh</b>.</li>
    </ul>
  </details>

  let saizTulisan = 18;

  function besarTulisan() {
    saizTulisan += 2;
    document.body.style.fontSize = saizTulisan + "px";
  }

  function kecilTulisan() {
    if (saizTulisan > 14) {
      saizTulisan -= 2;
      document.body.style.fontSize = saizTulisan + "px";
    }
  }

  function resetTulisan() {
    saizTulisan = 18;
    document.body.style.fontSize = saizTulisan + "px";
  }

  function semakAktiviti() {
    let markah = 0;

    let a1 = document.getElementById("a1").value.trim().toLowerCase();
    let a2 = document.getElementById("a2").value.trim().toLowerCase();
    let a3 = document.getElementById("a3").value.trim().toLowerCase();
    let a4 = document.getElementById("a4").value.trim().toLowerCase();

    if (a1 === "besar") markah++;
    if (a2 === "wangi") markah++;
    if (a3 === "gembira") markah++;
    if (a4 === "bulat") markah++;

    document.getElementById("keputusanAktiviti").innerHTML =
      "Markah kamu: " + markah + "/4<br>" +
      "Jawapan cadangan: 1. besar, 2. wangi, 3. gembira, 4. bulat";

    if (markah === 4) {
      document.getElementById("keputusanAktiviti").innerHTML +=
      "<br>🌟 Tahniah! Jawapan kamu sangat baik.";
    } else {
      document.getElementById("keputusanAktiviti").innerHTML +=
      "<br>😊 Cuba lagi. Kamu pasti boleh!";
    }
  }

  function semakKuiz() {
    let jawapanBetul = {
      q1: "B",
      q2: "C",
      q3: "C",
      q4: "A",
      q5: "B"
    };

    let markah = 0;
    let jumlah = 5;

    for (let q in jawapanBetul) {
      let pilihan = document.querySelector('input[name="' + q + '"]:checked');
      if (pilihan && pilihan.value === jawapanBetul[q]) {
        markah++;
      }
    }

    let mesej = "";

    if (markah === 5) {
      mesej = "Hebat! Kamu menguasai kata adjektif. 🌟";
    } else if (markah >= 3) {
      mesej = "Bagus! Teruskan usaha. 👍";
    } else {
      mesej = "Jangan risau. Baca nota dan cuba lagi. 😊";
    }

    document.getElementById("markahKuiz").innerHTML =
